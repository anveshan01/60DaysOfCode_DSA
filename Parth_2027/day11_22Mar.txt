class MinStack {

    Stack<Integer>st;
    PriorityQueue<Integer> pQueue;
            
    public MinStack() {
        st=new Stack<>();
        pQueue=new PriorityQueue<Integer>();

    }
    
    public void push(int val) {
        st.push(val);
        pQueue.add(val);
    }
    
    public void pop() {
        pQueue.remove(st.peek());
        st.pop();
        
    }
    
    public int top() {
        return st.peek();
    }
    
    public int getMin() {
        return pQueue.peek();
    }
}
