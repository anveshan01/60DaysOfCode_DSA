//50. Pow(x, n)


class Solution {
    double ans;
    public double myPow(double x, int n) {
        double ans=1;
        long nn=n;//to prevent - to + overflow
        if(n<0) nn=(-1)*nn;
        while(nn>0){
            if(nn%2==1){
                ans=ans*x;
                nn--;
            }else{
                    x=x*x;
                    nn/=2;
            }
        }
        if(n<0) ans=(double)(1.0)/ans;
        return ans;

    }
}
