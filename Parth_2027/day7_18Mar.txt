//9. Palindrome Number
class Solution {
    public boolean isPalindrome(int x) {
        if(x<10 && x>0) return true;
        if(x<0)return false;
        int y=x;
        int z=0;
        while(y!=0){
            int rem=y%10;
            z=z*10+rem;
            y=y/10;
        }
        return z==x;
    }
}
